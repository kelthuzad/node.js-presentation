Event Loop in Node.js

Ablauf

  • Ein Single Thread Event Loop verarbeitet alle auftretenden Events
  • Requests von Clients werden als Events angenommen
  • I/O Aufgaben werden als asynchrone Aufrufe deligiert (mit callbacks)
  • I/O Callbacks mit Ergebnissen werden als Events verarbeitet

WebService als Modul (V1)

Hello World als Grundlage Nun wird WebService als eigenes Modul ausgegliedert

Server-Modul (V1)

var http = require("http");

function start() {
    function onRequest(request, response) {
        response.writeHead(200, {"Content-Type": "text/plain"});
        response.write("Hello World");
        response.end();
    }

    http.createServer(onRequest).listen(8888);
}

exports.start = start;
Server nun als eigenes Modul (exports.XXX ist alles, was für das Modul benötigt wird) keine funktionalen Änderungen

index.js (V1)


var server = require("./server");

server.start();

index.js bindet nun das Server-Modul ein und startet es

keine Veränderung an der Funktionalität - > nur Modularisierung

Wer antwortet auf den
Request?

Problem bei blockierenden Funktionen allgemein: blockiert weitere ankommende Requests Probleme bei nicht blockierenden Funktionen, welche im Hintergrund laufen und durch ein Callback die Antwort übermitteln -> Return-Wert wird schon übermittelt, wenn Callback noch nicht wieder da ist -> Return ist empty

Request Handler
Funkionalität geben (V4)

Server-Modul (V4)

var http = require("http");      var url = require("url");

function start(route, handle) {
    function onRequest(request, response) {
        var postData = "";
        var pathname = url.parse(request.url).pathname;
        request.setEncoding("utf8");
        request.addListener("data", function (postDataChunk) {
            postData += postDataChunk;
        });
        request.addListener("end", function () {
            route(handle, pathname, response, postData);
        });
    }
    http.createServer(onRequest).listen(8888);
}
exports.start = start;
response ist nun nicht mehr im Server

Router-Modul (V4)

function route(handle, pathname, response, postData) {

    if (typeof handle[pathname] === 'function') {
        handle[pathname](response, postData);
    } else {
        response.writeHead(404, {"Content-Type": "text/plain"});
        response.write("404 Not found");
        response.end();
    }
}
exports.route = route;
