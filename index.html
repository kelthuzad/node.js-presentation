Parameter von onRequest()

  • request
  • response

können genauere Details des HTTP Requests liefern

request: Details zum eingehenden HTTP Request

response: Möglichkeit der Antwort auf den HTTP Request

bei "Hello World": request wird nicht gebraucht, es wird nur mit response auf den Request geantwortet

Server als eigenes Modul


var http = require("http");

function start() {
    function onRequest(request, response) {
        console.log("Request received.");
        response.writeHead(200, {"Content-Type": "text/plain"});
        response.write("Hello World");
        response.end();
    }

    http.createServer(onRequest).listen(8888);

    console.log("Server has started.");
}

exports.start = start;
